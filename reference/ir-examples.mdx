## Overview

This page shows actual intermediate representations (IR) generated by microMLC's compilation pipeline. Each stage demonstrates how the attention kernel is progressively transformed from high-level StableHLO operations to low-level vectorized memref code.

## Directory Structure

```
ir/
├── attn_complete.mlir           # Input: StableHLO from JAX
├── attn_stage0_linalg.mlir      # Stage 0: Linalg conversion
├── attn_stage1_fused.mlir       # Stage 1: Epilogue fusion
├── attn_stage2_tiled_tensor.mlir # Stage 2: Two-level tiling
├── attn_stage3_bufferized.mlir  # Stage 3: Bufferization
├── attn_stage5_vectorized.mlir  # Stage 5: Vectorization
└── attention_pipeline.mlir      # Complete pipeline trace
```

## Input: StableHLO Attention

**File**: `ir/attn_complete.mlir`

**Source**: Generated by `transformer/python/gen_mhattn_ir.py` using JAX's StableHLO export

This IR represents the mathematical structure of multi-head attention with explicit tensor operations:

```mlir
// --- MAP DEFINITIONS ---
#map_identity_4d = affine_map<(d0, d1, d2, d3) -> (d0, d1, d2, d3)>
#map_identity_3d = affine_map<(d0, d1, d2) -> (d0, d1, d2)>
#map_reduce_3d_2d = affine_map<(d0, d1, d2) -> (d0, d1)>

// Q transpose: (B, H, S, D) <- Input(B, S, H, D)
#map_transpose_q = affine_map<(d0, d1, d2, d3) -> (d0, d2, d1, d3)>

// K transpose: (B, H, D, S) <- Input(B, S, H, D)
#map_transpose_k = affine_map<(d0, d1, d2, d3) -> (d0, d3, d1, d2)>

module {
  func.func public @main(%arg0: tensor<1x128x128xf32>, 
                         %arg1: tensor<1x128x128xf32>, 
                         %arg2: tensor<1x128x128xf32>, 
                         %arg3: tensor<1x128x128xf32>) -> tensor<1x128x128xf32> {
    
    %cst_0 = arith.constant 0.000000e+00 : f32
    %cst_neg_inf = arith.constant 0xFF800000 : f32
    %cst_scale = arith.constant 0.176776695 : f32  // 1/sqrt(32)

    // ========================================================================
    // PART 1: LINEAR PROJECTIONS (Q, K, V)
    // ========================================================================
    %in_flat = tensor.collapse_shape %arg0 [[0, 1], [2]] 
               : tensor<1x128x128xf32> into tensor<128x128xf32>
    %w_q_flat = tensor.collapse_shape %arg1 [[0, 1], [2]] 
                : tensor<1x128x128xf32> into tensor<128x128xf32>
    %w_k_flat = tensor.collapse_shape %arg2 [[0, 1], [2]] 
                : tensor<1x128x128xf32> into tensor<128x128xf32>
    %w_v_flat = tensor.collapse_shape %arg3 [[0, 1], [2]] 
                : tensor<1x128x128xf32> into tensor<128x128xf32>

    %init_proj = tensor.empty() : tensor<128x128xf32>
    %q_proj = linalg.matmul ins(%in_flat, %w_q_flat : tensor<128x128xf32>, tensor<128x128xf32>) 
                            outs(%init_proj : tensor<128x128xf32>) -> tensor<128x128xf32>
    %k_proj = linalg.matmul ins(%in_flat, %w_k_flat : tensor<128x128xf32>, tensor<128x128xf32>) 
                            outs(%init_proj : tensor<128x128xf32>) -> tensor<128x128xf32>
    %v_proj = linalg.matmul ins(%in_flat, %w_v_flat : tensor<128x128xf32>, tensor<128x128xf32>) 
                            outs(%init_proj : tensor<128x128xf32>) -> tensor<128x128xf32>

    // ========================================================================
    // PART 2: HEAD SPLITTING (Reshape to 4 heads)
    // ========================================================================
    %q_expanded = tensor.expand_shape %q_proj [[0, 1], [2, 3]] 
                  : tensor<128x128xf32> into tensor<1x128x4x32xf32>
    %k_expanded = tensor.expand_shape %k_proj [[0, 1], [2, 3]] 
                  : tensor<128x128xf32> into tensor<1x128x4x32xf32>
    %v_expanded = tensor.expand_shape %v_proj [[0, 1], [2, 3]] 
                  : tensor<128x128xf32> into tensor<1x128x4x32xf32>

    %init_heads = tensor.empty() : tensor<1x4x128x32xf32>
    
    // Q: (B, S, H, D) -> (B, H, S, D)
    %q_heads = linalg.generic {
        indexing_maps = [#map_transpose_q, #map_identity_4d],
        iterator_types = ["parallel", "parallel", "parallel", "parallel"]
    } ins(%q_expanded : tensor<1x128x4x32xf32>) outs(%init_heads : tensor<1x4x128x32xf32>) {
        ^bb0(%in: f32, %out: f32): 
          linalg.yield %in : f32
    } -> tensor<1x4x128x32xf32>

    // ========================================================================
    // PART 3: SCALED DOT-PRODUCT ATTENTION
    // ========================================================================
    %q_flat = tensor.collapse_shape %q_heads [[0, 1], [2], [3]] 
              : tensor<1x4x128x32xf32> into tensor<4x128x32xf32>
    %k_flat = tensor.collapse_shape %k_heads_T [[0, 1], [2], [3]] 
              : tensor<1x4x32x128xf32> into tensor<4x32x128xf32>
    %v_flat = tensor.collapse_shape %v_heads [[0, 1], [2], [3]] 
              : tensor<1x4x128x32xf32> into tensor<4x128x32xf32>

    %scores_init = tensor.empty() : tensor<4x128x128xf32>
    %scores = linalg.batch_matmul ins(%q_flat, %k_flat : tensor<4x128x32xf32>, tensor<4x32x128xf32>)
                                  outs(%scores_init : tensor<4x128x128xf32>) -> tensor<4x128x128xf32>

    // Scaling by 1/sqrt(d_k)
    %scaled = linalg.generic {
        indexing_maps = [#map_identity_3d, #map_identity_3d],
        iterator_types = ["parallel", "parallel", "parallel"]
    } ins(%scores : tensor<4x128x128xf32>) outs(%scaled_init : tensor<4x128x128xf32>) {
        ^bb0(%in: f32, %out: f32):
            %res = arith.mulf %in, %cst_scale : f32
            linalg.yield %res : f32
    } -> tensor<4x128x128xf32>

    // Softmax: Max reduction
    %max = linalg.generic {
        indexing_maps = [#map_identity_3d, #map_reduce_3d_2d],
        iterator_types = ["parallel", "parallel", "reduction"]
    } ins(%scaled : tensor<4x128x128xf32>) outs(%max_fill : tensor<4x128xf32>) {
        ^bb0(%in: f32, %acc: f32):
            %gt = arith.cmpf ogt, %in, %acc : f32
            %sel = arith.select %gt, %in, %acc : f32
            linalg.yield %sel : f32
    } -> tensor<4x128xf32>

    // Softmax: Exp(x - max)
    %exp = linalg.generic {
        indexing_maps = [#map_identity_3d, #map_reduce_3d_2d, #map_identity_3d],
        iterator_types = ["parallel", "parallel", "parallel"]
    } ins(%scaled, %max : tensor<4x128x128xf32>, tensor<4x128xf32>) outs(%exp_init : tensor<4x128x128xf32>) {
        ^bb0(%in: f32, %in_max: f32, %out: f32):
            %sub = arith.subf %in, %in_max : f32
            %res = math.exp %sub : f32
            linalg.yield %res : f32
    } -> tensor<4x128x128xf32>

    // Softmax: Sum
    %sum = linalg.generic {
        indexing_maps = [#map_identity_3d, #map_reduce_3d_2d],
        iterator_types = ["parallel", "parallel", "reduction"]
    } ins(%exp : tensor<4x128x128xf32>) outs(%sum_fill : tensor<4x128xf32>) {
        ^bb0(%in: f32, %acc: f32):
            %add = arith.addf %in, %acc : f32
            linalg.yield %add : f32
    } -> tensor<4x128xf32>

    // Softmax: Normalize
    %probs = linalg.generic {
        indexing_maps = [#map_identity_3d, #map_reduce_3d_2d, #map_identity_3d],
        iterator_types = ["parallel", "parallel", "parallel"]
    } ins(%exp, %sum : tensor<4x128x128xf32>, tensor<4x128xf32>) outs(%prob_init : tensor<4x128x128xf32>) {
        ^bb0(%in: f32, %in_sum: f32, %out: f32):
            %div = arith.divf %in, %in_sum : f32
            linalg.yield %div : f32
    } -> tensor<4x128x128xf32>

    // Context vectors: probs @ V
    %context = linalg.batch_matmul ins(%probs, %v_flat : tensor<4x128x128xf32>, tensor<4x128x32xf32>)
                                   outs(%context_init : tensor<4x128x32xf32>) -> tensor<4x128x32xf32>

    // ========================================================================
    // PART 4: RECOMBINE HEADS
    // ========================================================================
    %expanded_10 = tensor.expand_shape %context [[0, 1], [2], [3]] 
                   : tensor<4x128x32xf32> into tensor<1x4x128x32xf32>
    
    %output = linalg.generic {
        indexing_maps = [#map_transpose_q, #map_identity_4d],
        iterator_types = ["parallel", "parallel", "parallel", "parallel"]
    } ins(%expanded_10 : tensor<1x4x128x32xf32>) outs(%output_init : tensor<1x128x4x32xf32>) {
        ^bb0(%in: f32, %out: f32): 
          linalg.yield %in : f32
    } -> tensor<1x128x4x32xf32>
    
    %collapsed_11 = tensor.collapse_shape %output [[0], [1], [2, 3]] 
                    : tensor<1x128x4x32xf32> into tensor<1x128x128xf32>
    
    return %collapsed_11 : tensor<1x128x128xf32>
  }
}
```

### Key Characteristics

- **Pure functional**: All operations are SSA (Single Static Assignment)
- **Named operations**: `linalg.matmul`, `linalg.batch_matmul` preserve semantics
- **Affine maps**: Explicit dimension reordering for transposes
- **Type safety**: All shapes tracked by type system

## Stage 0: Linalg Operations

**File**: `ir/attn_stage0_linalg.mlir`

**Transformation**: StableHLO → Linalg dialect

Excerpt showing softmax structure:

```mlir
// Batch matmul: Q @ K^T
%10 = linalg.batch_matmul ins(%collapsed_7, %collapsed_8 : tensor<4x128x32xf32>, tensor<4x32x128xf32>) 
                          outs(%9 : tensor<4x128x128xf32>) -> tensor<4x128x128xf32>

// Scale scores
%11 = linalg.generic {indexing_maps = [#map3, #map3], 
                      iterator_types = ["parallel", "parallel", "parallel"]} 
      ins(%10 : tensor<4x128x128xf32>) outs(%9 : tensor<4x128x128xf32>) {
  ^bb0(%in: f32, %out: f32):
    %23 = arith.mulf %in, %cst_1 : f32  // cst_1 = 0.176776692
    linalg.yield %23 : f32
} -> tensor<4x128x128xf32>

// Max reduction (for numerical stability)
%14 = linalg.generic {indexing_maps = [#map3, #map4], 
                      iterator_types = ["parallel", "parallel", "reduction"]} 
      ins(%11 : tensor<4x128x128xf32>) outs(%13 : tensor<4x128xf32>) {
  ^bb0(%in: f32, %out: f32):
    %23 = arith.cmpf ogt, %in, %out : f32
    %24 = arith.select %23, %in, %out : f32
    linalg.yield %24 : f32
} -> tensor<4x128xf32>

// Exp(score - max)
%15 = linalg.generic {indexing_maps = [#map3, #map4, #map3], 
                      iterator_types = ["parallel", "parallel", "parallel"]} 
      ins(%11, %14 : tensor<4x128x128xf32>, tensor<4x128xf32>) outs(%9 : tensor<4x128x128xf32>) {
  ^bb0(%in: f32, %in_12: f32, %out: f32):
    %23 = arith.subf %in, %in_12 : f32
    %24 = math.exp %23 : f32
    linalg.yield %24 : f32
} -> tensor<4x128x128xf32>

// Sum reduction
%17 = linalg.generic {indexing_maps = [#map3, #map4], 
                      iterator_types = ["parallel", "parallel", "reduction"]} 
      ins(%15 : tensor<4x128x128xf32>) outs(%16 : tensor<4x128xf32>) {
  ^bb0(%in: f32, %out: f32):
    %23 = arith.addf %in, %out : f32
    linalg.yield %23 : f32
} -> tensor<4x128xf32>

// Normalize: exp / sum
%18 = linalg.generic {indexing_maps = [#map3, #map4, #map3], 
                      iterator_types = ["parallel", "parallel", "parallel"]} 
      ins(%15, %17 : tensor<4x128x128xf32>, tensor<4x128xf32>) outs(%9 : tensor<4x128x128xf32>) {
  ^bb0(%in: f32, %in_12: f32, %out: f32):
    %23 = arith.divf %in, %in_12 : f32
    linalg.yield %23 : f32
} -> tensor<4x128x128xf32>
```

**Analysis**: 5 separate generic operations for softmax, each with its own allocation.

## Stage 1: Epilogue Fusion

**File**: `ir/attn_stage1_fused.mlir`

**Transformation**: Fuse elementwise ops into producers

Fused softmax computation (compare to Stage 0):

```mlir
// Batch matmul: Q @ K^T (unchanged)
%10 = linalg.batch_matmul ins(%collapsed_7, %collapsed_8 : tensor<4x128x32xf32>, tensor<4x32x128xf32>) 
                          outs(%9 : tensor<4x128x128xf32>) -> tensor<4x128x128xf32>

// Max reduction (unchanged)
%14 = linalg.generic {indexing_maps = [#map3, #map4], 
                      iterator_types = ["parallel", "parallel", "reduction"]} 
      ins(%11 : tensor<4x128x128xf32>) outs(%13 : tensor<4x128xf32>) {
  ^bb0(%in: f32, %out: f32):
    %23 = arith.cmpf ogt, %in, %out : f32
    %24 = arith.select %23, %in, %out : f32
    linalg.yield %24 : f32
} -> tensor<4x128xf32>

// FUSED: scale + exp (was 2 ops, now 1)
%15 = linalg.generic {indexing_maps = [#map3, #map4, #map3], 
                      iterator_types = ["parallel", "parallel", "parallel"]} 
      ins(%10, %14 : tensor<4x128x128xf32>, tensor<4x128xf32>) outs(%9 : tensor<4x128x128xf32>) {
  ^bb0(%in: f32, %in_12: f32, %out: f32):
    %23 = arith.mulf %in, %cst_1 : f32        // scale (was separate op)
    %24 = arith.subf %23, %in_12 : f32        // subtract max
    %25 = math.exp %24 : f32                   // exp (was separate op)
    linalg.yield %25 : f32
} -> tensor<4x128x128xf32>

// Sum reduction (unchanged)
%17 = linalg.generic {indexing_maps = [#map3, #map4], 
                      iterator_types = ["parallel", "parallel", "reduction"]} 
      ins(%15 : tensor<4x128x128xf32>) outs(%16 : tensor<4x128xf32>) {
  ^bb0(%in: f32, %out: f32):
    %23 = arith.addf %in, %out : f32
    linalg.yield %23 : f32
} -> tensor<4x128xf32>

// FUSED: scale + exp + div (was 3 ops, now 1)
%18 = linalg.generic {indexing_maps = [#map3, #map4, #map4, #map3], 
                      iterator_types = ["parallel", "parallel", "parallel"]} 
      ins(%10, %14, %17 : tensor<4x128x128xf32>, tensor<4x128xf32>, tensor<4x128xf32>) 
      outs(%9 : tensor<4x128x128xf32>) {
  ^bb0(%in: f32, %in_12: f32, %in_13: f32, %out: f32):
    %23 = arith.mulf %in, %cst_1 : f32        // scale (recomputed)
    %24 = arith.subf %23, %in_12 : f32        // subtract max (recomputed)
    %25 = math.exp %24 : f32                   // exp (recomputed)
    %26 = arith.divf %25, %in_13 : f32        // div (was separate)
    linalg.yield %26 : f32
} -> tensor<4x128x128xf32>
```

**Benefit**: Reduced from 5 ops to 3 ops. Eliminated 2 intermediate allocations (16KB each).

**Trade-off**: `exp` is recomputed in the final op, but this is faster than reading from memory.

## Stage 2: Two-Level Tiling

**File**: `ir/attn_stage2_tiled_tensor.mlir` (excerpt)

Tiling transforms flat operations into nested loops:

```mlir
// Original: %10 = linalg.batch_matmul ins(%q, %k) outs(%scores)

// Tiled version:
scf.for %arg5 = %c0 to %c4 step %c1 {  // Batch dimension
  // L2 tile loops (64x64)
  scf.for %arg6 = %c0 to %c128 step %c64 {  // M dimension
    scf.for %arg7 = %c0 to %c128 step %c64 {  // N dimension
      scf.for %arg8 = %c0 to %c32 step %c32 {  // K dimension (no tiling, fits in cache)
        
        // Extract L2 tiles
        %tile_q = tensor.extract_slice %q_flat[%arg5, %arg6, %arg8] [1, 64, 32] [1, 1, 1] 
                  : tensor<4x128x32xf32> to tensor<1x64x32xf32>
        %tile_k = tensor.extract_slice %k_flat[%arg5, %arg8, %arg7] [1, 32, 64] [1, 1, 1] 
                  : tensor<4x32x128xf32> to tensor<1x32x64xf32>
        %tile_out = tensor.extract_slice %scores[%arg5, %arg6, %arg7] [1, 64, 64] [1, 1, 1] 
                    : tensor<4x128x128xf32> to tensor<1x64x64xf32>
        
        // L1 tile loops (4x4)
        scf.for %arg9 = %c0 to %c64 step %c4 {  // M inner
          scf.for %arg10 = %c0 to %c64 step %c4 {  // N inner
            scf.for %arg11 = %c0 to %c32 step %c4 {  // K inner
              
              // Extract L1 micro-tiles
              %micro_q = tensor.extract_slice %tile_q[0, %arg9, %arg11] [1, 4, 4] [1, 1, 1] 
                         : tensor<1x64x32xf32> to tensor<1x4x4xf32>
              %micro_k = tensor.extract_slice %tile_k[0, %arg11, %arg10] [1, 4, 4] [1, 1, 1] 
                         : tensor<1x32x64xf32> to tensor<1x4x4xf32>
              %micro_out = tensor.extract_slice %tile_out[0, %arg9, %arg10] [1, 4, 4] [1, 1, 1] 
                           : tensor<1x64x64xf32> to tensor<1x4x4xf32>
              
              // 4x4x4 matmul (vectorizable)
              %result = linalg.batch_matmul ins(%micro_q, %micro_k : tensor<1x4x4xf32>, tensor<1x4x4xf32>) 
                                            outs(%micro_out : tensor<1x4x4xf32>) -> tensor<1x4x4xf32>
              
              // Insert result back
              %updated = tensor.insert_slice %result into %tile_out[0, %arg9, %arg10] [1, 4, 4] [1, 1, 1] 
                         : tensor<1x4x4xf32> into tensor<1x64x64xf32>
            }
          }
        }
      }
    }
  }
}
```

**Dimensions**:
- Outer loops: 4 batches × 2 (128/64) × 2 (128/64) = 16 L2 tiles
- Inner loops: 16 (64/4) × 16 (64/4) × 8 (32/4) = 2048 L1 tiles per L2 tile
- Total: 32,768 micro-matmuls of size 4×4×4

**Cache efficiency**:
- L2 tile: 64×32 + 32×64 + 64×64 = 2048 + 2048 + 4096 = 8192 floats = 32KB
- L1 tile: 4×4 + 4×4 + 4×4 = 16 + 16 + 16 = 48 floats = 192 bytes

## Stage 3: Bufferization

**File**: `ir/attn_stage3_bufferized.mlir` (excerpt)

Tensors become memrefs with explicit allocation:

```mlir
func.func public @main(%arg0: memref<1x128x128xf32, strided<[?, ?, ?], offset: ?>>, 
                       %arg1: memref<1x128x128xf32, strided<[?, ?, ?], offset: ?>>, 
                       %arg2: memref<1x128x128xf32, strided<[?, ?, ?], offset: ?>>, 
                       %arg3: memref<1x128x128xf32, strided<[?, ?, ?], offset: ?>>) -> memref<1x128x128xf32> {
  %cst = arith.constant 0.000000e+00 : f32
  %cst_0 = arith.constant 0xFF800000 : f32
  %cst_1 = arith.constant 0.176776692 : f32
  
  // Allocations replace tensor.empty()
  %alloc = memref.alloc() : memref<128x128xf32>
  %alloc_2 = memref.alloc() : memref<128x128xf32>
  %alloc_3 = memref.alloc() : memref<128x128xf32>
  
  // Collapse becomes reinterpret_cast
  %reinterpret = memref.reinterpret_cast %arg0 to 
                 offset: [0], sizes: [128, 128], strides: [128, 1] 
                 : memref<1x128x128xf32, strided<[?, ?, ?], offset: ?>> to memref<128x128xf32>
  
  // Matmul writes directly to allocation
  linalg.matmul ins(%reinterpret, %reinterpret_2 : memref<128x128xf32>, memref<128x128xf32>) 
                outs(%alloc : memref<128x128xf32>)
  
  // Expand becomes another reinterpret_cast
  %expand = memref.expand_shape %alloc [[0, 1], [2, 3]] output_shape [1, 128, 4, 32] 
            : memref<128x128xf32> into memref<1x128x4x32xf32>
  
  // Generic ops become in-place
  scf.for %arg5 = %c0 to %c4 step %c1 {
    scf.for %arg6 = %c0 to %c128 step %c64 {
      %subview = memref.subview %alloc_scores[%arg5, %arg6, 0] [1, 64, 128] [1, 1, 1] 
                 : memref<4x128x128xf32> to memref<1x64x128xf32, strided<[?, ?, ?], offset: ?>>
      
      linalg.generic {indexing_maps = [#map, #map], 
                      iterator_types = ["parallel", "parallel", "parallel"]} 
                     ins(%subview : memref<1x64x128xf32, strided<[?, ?, ?], offset: ?>>) 
                     outs(%subview : memref<1x64x128xf32, strided<[?, ?, ?], offset: ?>>) {
        ^bb0(%in: f32, %out: f32):
          %scaled = arith.mulf %in, %cst_1 : f32
          linalg.yield %scaled : f32
      }
    }
  }
  
  // Return allocated buffer (caller must dealloc)
  %result = memref.alloc() : memref<1x128x128xf32>
  return %result : memref<1x128x128xf32>
}
```

**Key changes**:
- `tensor<...>` → `memref<..., strided<...>>`
- `tensor.extract_slice` → `memref.subview` (no copy, just pointer arithmetic)
- `tensor.insert_slice` → removed (writes are in-place)
- `tensor.empty()` → `memref.alloc()` (actual allocation)

## Stage 5: Vectorization

**File**: `ir/attn_stage5_vectorized.mlir` (excerpt)

Scalar loops become vector operations:

```mlir
scf.for %arg9 = %c0 to %c64 step %c4 {  // L1 M loop
  scf.for %arg10 = %c0 to %c64 step %c4 {  // L1 N loop
    scf.for %arg11 = %c0 to %c32 step %c4 {  // L1 K loop
      
      // Load 4x4 tiles as vectors
      %vq_0 = vector.transfer_read %micro_q[%c0, %c0, %arg11], %pad 
              : memref<1x4x32xf32>, vector<4xf32>
      %vq_1 = vector.transfer_read %micro_q[%c0, %c1, %arg11], %pad 
              : memref<1x4x32xf32>, vector<4xf32>
      %vq_2 = vector.transfer_read %micro_q[%c0, %c2, %arg11], %pad 
              : memref<1x4x32xf32>, vector<4xf32>
      %vq_3 = vector.transfer_read %micro_q[%c0, %c3, %arg11], %pad 
              : memref<1x4x32xf32>, vector<4xf32>
      
      %vk_0 = vector.transfer_read %micro_k[%c0, %arg11, %c0], %pad 
              : memref<1x32x4xf32>, vector<4xf32>
      %vk_1 = vector.transfer_read %micro_k[%c0, %arg11, %c1], %pad 
              : memref<1x32x4xf32>, vector<4xf32>
      %vk_2 = vector.transfer_read %micro_k[%c0, %arg11, %c2], %pad 
              : memref<1x32x4xf32>, vector<4xf32>
      %vk_3 = vector.transfer_read %micro_k[%c0, %arg11, %c3], %pad 
              : memref<1x32x4xf32>, vector<4xf32>
      
      %vacc_0 = vector.transfer_read %micro_out[%c0, %c0, %c0], %zero 
                : memref<1x4x4xf32>, vector<4xf32>
      %vacc_1 = vector.transfer_read %micro_out[%c0, %c1, %c0], %zero 
                : memref<1x4x4xf32>, vector<4xf32>
      %vacc_2 = vector.transfer_read %micro_out[%c0, %c2, %c0], %zero 
                : memref<1x4x4xf32>, vector<4xf32>
      %vacc_3 = vector.transfer_read %micro_out[%c0, %c3, %c0], %zero 
                : memref<1x4x4xf32>, vector<4xf32>
      
      // 4x4 matrix multiply using FMA (Fused Multiply-Add)
      // Row 0: vacc_0 += vq_0 * vk
      %v0_0 = vector.fma %vq_0, %vk_0, %vacc_0 : vector<4xf32>
      %v0_1 = vector.fma %vq_0, %vk_1, %v0_0 : vector<4xf32>
      %v0_2 = vector.fma %vq_0, %vk_2, %v0_1 : vector<4xf32>
      %v0_3 = vector.fma %vq_0, %vk_3, %v0_2 : vector<4xf32>
      
      // Row 1: vacc_1 += vq_1 * vk
      %v1_0 = vector.fma %vq_1, %vk_0, %vacc_1 : vector<4xf32>
      %v1_1 = vector.fma %vq_1, %vk_1, %v1_0 : vector<4xf32>
      %v1_2 = vector.fma %vq_1, %vk_2, %v1_1 : vector<4xf32>
      %v1_3 = vector.fma %vq_1, %vk_3, %v1_2 : vector<4xf32>
      
      // ... rows 2 and 3 similar ...
      
      // Store results
      vector.transfer_write %v0_3, %micro_out[%c0, %c0, %c0] 
                            : vector<4xf32>, memref<1x4x4xf32>
      vector.transfer_write %v1_3, %micro_out[%c0, %c1, %c0] 
                            : vector<4xf32>, memref<1x4x4xf32>
      vector.transfer_write %v2_3, %micro_out[%c0, %c2, %c0] 
                            : vector<4xf32>, memref<1x4x4xf32>
      vector.transfer_write %v3_3, %micro_out[%c0, %c3, %c0] 
                            : vector<4xf32>, memref<1x4x4xf32>
    }
  }
}
```

**SIMD mapping**:
- Each `vector<4xf32>` maps to a single 128-bit register (SSE/NEON/AVX128)
- `vector.fma` compiles to a single CPU instruction (VFMADD on x86, FMLA on ARM)
- 16 vector loads + 64 FMAs + 4 stores = 84 vector instructions per micro-tile
- Compare to scalar: 64 loads + 64 multiplies + 64 adds + 16 stores = 208 scalar instructions

**Vectorization benefit**: ~2.5× instruction reduction + hardware parallelism = 3-4× speedup

## Reduction Vectorization Example

Softmax sum reduction (from Stage 5):

```mlir
// Scalar loop (before vectorization):
scf.for %i = %c0 to %c128 step %c1 {
  %val = memref.load %exp[%b, %row, %i] : memref<4x128x128xf32>
  %sum = arith.addf %sum, %val : f32
}

// Vectorized loop:
%vzero = arith.constant dense<0.0> : vector<4xf32>
%vacc = %vzero : vector<4xf32>
scf.for %i = %c0 to %c128 step %c4 {
  %vval = vector.transfer_read %exp[%b, %row, %i], %zero 
          : memref<4x128x128xf32>, vector<4xf32>
  %vacc_new = arith.addf %vacc, %vval : vector<4xf32>
}
// Horizontal reduction: sum all 4 lanes
%sum = vector.reduction <add>, %vacc_new : vector<4xf32> into f32
```

**Hardware instruction**: `vredsum.vs` (RISC-V), `VADDV` (ARM NEON), `haddps` (x86 SSE3)
